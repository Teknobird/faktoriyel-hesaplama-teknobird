function kopyala() { // Sonucun yazılı olduğu HTML elemanını seçin (kendi ID'niz ile güncelleyin, örn: 'sonuc-alani') const sonucMetni = document.getElementById('sonuc-alani').innerText || document.getElementById('sonuc-alani').value; const kopyalaButon = document.getElementById('kopyala-buton'); // Butonunuzun ID'si if (!sonucMetni) { alert("Kopyalanacak bir sonuç bulunamadı!"); return; } // Modern Clipboard API kullanımı (iframe'de allow="clipboard-write" gerektirir) if (navigator.clipboard && window.isSecureContext) { navigator.clipboard.writeText(sonucMetni) .then(() => { butonGeriBildirim(kopyalaButon); }) .catch(err => { console.error('Kopyalama hatası: ', err); alternatifKopyalama(sonucMetni, kopyalaButon); }); } else { // Tarayıcı desteklemiyorsa veya güvenli bağlantı (HTTPS) değilse alternatif yöntem alternatifKopyalama(sonucMetni, kopyalaButon); } } // Eski tarayıcılar ve fallback durumları için alternatif kopyalama yöntemi function alternatifKopyalama(text, buton) { const textArea = document.createElement("textarea"); textArea.value = text; // Görünmez yapıp sayfaya ekliyoruz textArea.style.position = "fixed"; textArea.style.left = "-999999px"; document.body.appendChild(textArea); textArea.select(); textArea.setSelectionRange(0, 99999); // Mobil için try { document.execCommand('copy'); butonGeriBildirim(buton); } catch (err) { alert('Kopyalanamadı, lütfen metni manuel seçerek kopyalayın.'); } document.body.removeChild(textArea); } // Kullanıcıya başarılı kopyalama bildirimini gösteren yardımcı fonksiyon function butonGeriBildirim(buton) { if (!buton) return; const eskiMetin = buton.innerHTML; buton.innerHTML = "✓ Kopyalandı!"; buton.style.backgroundColor = "#2ecc71"; // Geçici yeşil renk (isteğe bağlı) setTimeout(() => { buton.innerHTML = eskiMetin; buton.style.backgroundColor = ""; // Eski rengine geri döner }, 2000); }
